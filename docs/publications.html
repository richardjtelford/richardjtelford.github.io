# Publications

```{r load}
library("RefManageR")
source("R/bibstyler.R")
bib <- ReadBib("resources/works.bib")

for(i in 1:length(bib)){
  bib[[i]]$pages <- gsub("(?<!-)-(?!-)", "--", gsub(" ", "", bib[[i]]$pages), perl = TRUE)
  if(!is.null(bib[[i]]$doi)){
      bib[[i]]$doi <- gsub(".*\\.org/", "", bib[[i]]$doi)
    bib[[i]]$url <- NULL
  }
}

```
